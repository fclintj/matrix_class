250 Mat conv2d(Mat X, Mat H) {
251     Mat Y(X.m,X.n);
252     for (int i = 0; i < X.m; i++) {
253         for (int j = 0; j < X.n; j++) {
254             int sum1 = 0;
255             int sum2 = 0;
256 
257             // calculate convolution for each pixel
258             for (int k = 0; k < 3; k++) {
259                 for (int l = 0; l < 3; l++) {
260                     if ((i+k-1 > 0 && j+l-1 > 0) && (i+k-1 < X.m && j+l-1 < X.n)) {
261                         sum1 += X.loc[i+k-1][j+l] * H.loc[k][l];
262                         sum2 += X.loc[i+k-1][j+l] * H.loc[k][l];
263                     }
264                 }
265             }
266             Y.loc[i][j] = sqrt(sum1*sum1 + sum2*sum2);
267         }
268     }
269     return Y;
270 }
